Array

operating_systems = ['Fedora', 'SuSE', 'RHEL', 'Windows', 'MacOS']
linux_systems = ['RHEL', 'SuSE', 'PCLinuxOS', 'Ubuntu', 'Fedora']

# Union
operating_systems | linux_systems
# => ["Fedora", "SuSE", "RHEL", "Windows", "MacOS", "PCLinuxOS", "Ubuntu"] 

# Intersection
operating_systems & linux_systems      # => ["Fedora", "SuSE", "RHEL"] 

# Difference
operating_systems - linux_systems      # => ["Windows", "MacOS"] 
linux_systems - operating_systems      # => ["PCLinuxOS", "Ubuntu"] 

# Uniq
operating_systems = ['Fedora', 'SuSE', 'MacOS', 'RHEL', 'SuSE', 'Fedora']
operating_systems.uniq                 # => ["Fedora", "SuSE", "MacOS", "RHEL"]
